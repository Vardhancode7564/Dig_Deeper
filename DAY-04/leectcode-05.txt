class Solution {
public:
    string longestPalindrome(string s) {
        int start=0,maxLen=0;
        int n=s.size();

        for (int i=0; i<n;i++) {
            checkpalin(s,i,i,start,maxLen);    
            checkpalin(s,i,i+1,start,maxLen); 
        }

        return s.substr(start,maxLen);
    }

private:
    void checkpalin(const string& s,int left,int right, int& start, int& maxLen) {
        while (left>=0 && right<s.size() && s[left]==s[right]) {
            left--;
            right++;
        }
        int len=right-left-1;
        if(len>maxLen) {
            maxLen=len;
            start=left+1;
        }
    }
};
