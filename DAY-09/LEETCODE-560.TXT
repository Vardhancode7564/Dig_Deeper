class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        unordered_map<int,int>mpp;
        mpp[0]=1;

        int prefixSum=0, count=0;
        for (int num:nums) {
            prefixSum+=num;
            if (mpp.count(prefixSum-k)) {
                count+=mpp[prefixSum - k];
            }
            mpp[prefixSum]++;
        }
        return count;
    }
};
