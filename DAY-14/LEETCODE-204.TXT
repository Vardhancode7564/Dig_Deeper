class Solution {
public:
    int countPrimes(int n) {
        if(n<2) return 0;
        vector<int>primes;
        int count=0;
        for(int num=2;num<n;num++){
            bool isprime=true;
            for(int prime:primes){
                if(prime*prime>num) break;
                if(num%prime==0){
                    isprime=false;
                    break;
                }
            }
            if(isprime){
                primes.push_back(num);
                count++;
            }
        }
        return count;
    }
};
