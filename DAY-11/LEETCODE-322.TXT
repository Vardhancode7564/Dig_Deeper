class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        const int var=1e9;
        vector<int>dp(amount+1,var);
        dp[0]=0;

        for(int a=1;a<=amount;a++) {
            for(int c:coins) {
                if (a-c>=0) {
                    dp[a]=min(dp[a],dp[a-c]+1);
                }
            }
        }

        return dp[amount]==var?-1:dp[amount];
    }
};
